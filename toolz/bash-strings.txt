${string#substring}	# ДП РЕТЧПЗП У ОБЮБМБ
${string##substring}	# ДП РПУМЕДОЕЗП У ОБЮБМБ
${string%substring}	# ДП РЕТЧПЗП У ЛПОГБ
${string%%substring}	# ДП РПУМЕДОЕЗП У ЛПОГБ
